def scan_plan_files(self, include_backup=False):
    """扫描计划打印目录中的Excel文件
    
    Args:
        include_backup: 是否同时扫描backup目录（用于重新打印）
    """
    existing_plans = {}  # plan_no -> file_path
    
    # 扫描的目录列表
    dirs_to_scan = []
    if os.path.exists(self.plan_dir):
        dirs_to_scan.append(self.plan_dir)
    
    # 只有需要重新打印时才扫描backup目录
    if include_backup:
        backup_dir = os.path.join(self.plan_dir, "backup")
        if os.path.exists(backup_dir):
            dirs_to_scan.append(backup_dir)
    
    for scan_dir in dirs_to_scan:
        for file in os.listdir(scan_dir):
            file_path = os.path.join(scan_dir, file)
            # 跳过目录
            if os.path.isdir(file_path):
                print(f"跳过目录: {file}")
                continue
            
            # 跳过装炉顺序文件和合并文件
            if file == "装炉顺序.xls" or file == "装炉顺序_显示.xls" or file == "合并文件.xls":
                print(f"跳过系统文件: {file}")
                continue
            
            if file.endswith(".xls") or file.endswith(".xlsx"):
                pass
            else:
                continue
            try:
                # 打开文件读取计划号和粗轧报信字段
                success, result = self.safe_read_excel(file_path, max_retries=3, retry_delay=0.5)
                if not success:
                    print(f"读取文件失败 {file}: {result}")
                    continue
                workbook = result
                sheet = workbook.sheet_by_index(0)
                
                if sheet.nrows < 1:
                    continue
                
                # 查找计划号列和粗轧报信列
                plan_col = -1
                baoxin_col = -1
                header_row = -1
                
                # 在前10行中查找表头（处理后的文件表头可能在更下面）
                for row in range(min(10, sheet.nrows)):
                    for col in range(sheet.ncols):
                        cell_value = str(sheet.cell_value(row, col)).strip()
                        if cell_value == "计划号":
                            plan_col = col
                            header_row = row
                        elif "粗轧报信" in cell_value or "粗轧" in cell_value:
                            baoxin_col = col
                
                # 如果没有找到粗轧报信列，跳过此文件
                if baoxin_col == -1:
                    print(f"文件 {file} 中没有粗轧报信字段，跳过")
                    continue
                
                # 读取计划号
                # 首先尝试从文件名获取计划号（因为文件名已经是计划号）
                file_name_without_ext = os.path.splitext(file)[0]
                plan_no_from_file = file_name_without_ext.strip()
                
                # 然后尝试从文件内容读取计划号（检查多行，因为处理后的文件可能有空行）
                plan_no_from_content = ""
                if plan_col != -1:
                    # 检查header_row之后的几行，找到非空的计划号
                    for check_row in range(header_row + 1, min(header_row + 5, sheet.nrows)):
                        cell_value = sheet.cell_value(check_row, plan_col)
                        # 处理可能的浮点数格式（如 12345.0）
                        if isinstance(cell_value, float):
                            if cell_value == int(cell_value):
                                plan_no_from_content = str(int(cell_value)).strip()
                            else:
                                plan_no_from_content = str(cell_value).strip()
                        else:
                            plan_no_from_content = str(cell_value).strip()
                        
                        if plan_no_from_content and plan_no_from_content != "":
                            break
                
                # 优先使用文件内容中的计划号，如果没有则使用文件名
                plan_no = plan_no_from_content if plan_no_from_content else plan_no_from_file
                
                if plan_no and plan_no != "":
                    # 只在主目录中进行文件重命名，backup目录中的文件不重命名
                    if scan_dir == self.plan_dir:
                        # 检查文件名是否已经是计划号
                        if file_name_without_ext != plan_no:
                            # 需要重命名文件
                            file_ext = os.path.splitext(file)[1]
                            new_file_name = plan_no + file_ext
                            new_file_path = os.path.join(self.plan_dir, new_file_name)
                            
                            # 检查目标文件是否已存在
                            if os.path.exists(new_file_path):
                                print(f"文件 {file} 需要重命名为 {new_file_name}，但目标文件已存在，跳过重命名")
                                existing_plans[plan_no] = file_path
                            else:
                                try:
                                    os.rename(file_path, new_file_path)
                                    print(f"文件 {file} 已重命名为 {new_file_name}")
                                    existing_plans[plan_no] = new_file_path
                                except Exception as rename_error:
                                    print(f"重命名文件 {file} 失败: {str(rename_error)}")
                                    existing_plans[plan_no] = file_path
                        else:
                            # 文件名已经是计划号，无需重命名
                            existing_plans[plan_no] = file_path
                            print(f"文件 {file} -> 计划号: {plan_no}")
                    else:
                        # backup目录中的文件，直接记录映射关系
                        existing_plans[plan_no] = file_path
                        print(f"[Backup] 文件 {file} -> 计划号: {plan_no}")
                
                # 释放workbook资源
                try:
                    workbook.release_resources()
                except:
                    pass
                
            except Exception as e:
                print(f"读取文件 {file} 失败: {str(e)}")
                # 确保释放资源
                try:
                    if 'workbook' in locals():
                        workbook.release_resources()
                except:
                    pass
                continue
    
    return existing_plans
